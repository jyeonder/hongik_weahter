    async function getWeather() {

      try {

        const res = await fetch("https://api.open-meteo.com/v1/forecast?latitude=37.5665&longitude=126.9780&current_weather=true&hourly=temperature_2m,weathercode&daily=temperature_2m_max,weathercode&timezone=Asia%2FSeoul");

        const data = await res.json();

        const now = new Date();

        document.getElementById("weekday").textContent = now.toLocaleDateString("en-US", { weekday: "long" });

        document.getElementById("temperature").textContent = ${Math.round(data.current_weather.temperature)}°;

        const weatherCode = data.current_weather.weathercode;

        const description = weatherCode < 3 ? "Clear" : weatherCode < 45 ? "Cloudy" : "Rainy";

        document.getElementById("desc-text").textContent = description;

        document.getElementById("weather-icon").innerHTML = weatherCode < 3

          ? '<circle cx="12" cy="12" r="9" fill="none" stroke="#222" stroke-width="3" />'

          : '<path d="M5 16h14M7 14a7 7 0 0 1 10 0" fill="none" stroke="#222" stroke-width="3" />';


        const hourlyContainer = document.getElementById("hourly");

        const nowHour = now.getHours();

        for (let i = 0; i < 5; i++) {

          const index = data.hourly.time.findIndex(t => new Date(t).getHours() === nowHour + i);

          if (index !== -1) {

            const hour = nowHour + i;

            const temp = Math.round(data.hourly.temperature_2m[index]);

            const item = document.createElement("div");

            item.className = "hour";

            item.innerHTML = <p>${hour}:00</p><p>${temp}°</p>;

            hourlyContainer.appendChild(item);

          }

        }


        const forecastContainer = document.getElementById("forecast");

        data.daily.time.slice(0, 6).forEach((dateStr, i) => {

          const date = new Date(dateStr);

          const dayName = date.toLocaleDateString("en-US", { weekday: "short" });

          const temp = Math.round(data.daily.temperature_2m_max[i]);

          const icon = data.daily.weathercode[i] < 3

            ? '<circle cx="9" cy="9" r="7" stroke="#222" stroke-width="2" fill="none" />'

            : '<path d="M3 11h12M5 9a6 6 0 0 1 10 0" stroke="#222" stroke-width="2" fill="none" />';

          const div = document.createElement("div");

          div.className = "day";

          div.innerHTML = <h4>${dayName}</h4><svg viewBox='0 0 18 18'>${icon}</svg><p>${temp}°</p>;

          forecastContainer.appendChild(div);

        });

      } catch (err) {

        document.getElementById("desc-text").textContent = "Failed to load weather.";

      }

    }


    function createClock(city, tz) {

      const container = document.createElement("div");

      const clock = document.createElement("div");

      clock.className = "clock";

      const label = document.createElement("div");

      label.className = "clock-label";

      label.textContent = city;


      const hourHand = document.createElement("div");

      hourHand.className = "hand hour-hand";

      const minuteHand = document.createElement("div");

      minuteHand.className = "hand minute-hand";

      const secondHand = document.createElement("div");

      secondHand.className = "hand second-hand";

      clock.append(hourHand, minuteHand, secondHand);

      container.append(clock, label);

      document.getElementById("clocks").appendChild(container);


      function updateClock() {

        const now = new Date(new Date().toLocaleString("en-US", { timeZone: tz }));

        const h = now.getHours();

        const m = now.getMinutes();

        const s = now.getSeconds();

        const isNight = h < 6 || h >= 18;

        if (isNight) clock.classList.add("dark");

        else clock.classList.remove("dark");

        hourHand.style.transform = rotate(${(h % 12) * 30 + m * 0.5}deg);

        minuteHand.style.transform = rotate(${m * 6}deg);

        secondHand.style.transform = rotate(${s * 6}deg);

      }

      updateClock();

      setInterval(updateClock, 1000);

    }


    getWeather();

      { name: "Paris", tz: "Europe/Paris" },

      { name: "Tokyo", tz: "Asia/Tokyo" },

      { name: "Los Angeles", tz: "America/Los_Angeles" }

    ];

    cities.forEach(c => createClock(c.name, c.tz));
